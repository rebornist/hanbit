# parcel 세팅
1. yarn 설치 : https://yarnpkg.com/lang/en/docs/install/#windows-stable
2. yarn init
3. yarn global add parcel-bundler node-sass

# tailwindcss 세팅
1. yarn add babel-preset-env bable-preset-es2015 tailwindcss postcss-modules postcss-cli
2. postcss.config.js 생성 후
module.exports = {
  plugins: [require("tailwindcss")]
};
추가
3. styles.scss 생성 후 @tailwind base; @tailwind components; @tailwind utilities; 추가
4. package.json "scripts"에 {"build:css": "postcss styles.scss -o ../css/tailwind.css"} 추가
   또는 base.js 생성 후 import tailwindcss from "./styles.scss";

# bootstrap 세팅
1. yarn add jquery popper.js bootstrap
2. ackage.json "scripts"에 {"build": "parcel build base.js -d ../js"} 추가
3. base.js 생성 후 import $ from "jquery"; import popper from "popper.js"; import bootstrap from "bootstrap"; 추가
